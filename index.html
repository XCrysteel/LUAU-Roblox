Inventory Manager
Module

A lightweight Lua module for managing player inventory tables with automated stacking and sorting functionality. 

local Inventory = {}
Inventory.__index = Inventory

function Inventory.new(maxSlots)
    local self = setmetatable({}, Inventory)
    self.items = {}
    self.maxSlots = maxSlots or 20
    return self
end

function Inventory:addItem(itemName, amount)
    if #self.items >= self.maxSlots then
        print("Inventory Full!")
        return false
    end
    
    table.insert(self.items, {name = itemName, count = amount})
    return true
end

return Inventory
NPC Pathfinding
AI Script

Simple implementation of A* pathfinding for NPCs in a grid-based environment. 

-- Simple Distance Check
local function getDistance(p1, p2)
    return math.sqrt((p1.x - p2.x)^2 + (p1.y - p2.y)^2)
end

local function moveToTarget(npc, target)
    local dist = getDistance(npc.Position, target.Position)
    
    if dist > 5 then
        npc:MoveTo(target.Position)
        print("Moving to target...")
    else
        print("Target reached.")
    end
end
